Any operation would modify the original array. 
                var Arr = [1, 2, 3, 4, 5];

                console.log(Arr);
                console.log(Arr.length);
    
                console.log("-------------------SOME-------------------------------");
                let x = Arr.some(x => x % 2 == 0);
                console.log(x);
    
                console.log("-------------------EVERY-------------------------------");
                let y = Arr.every(x => x % 2 == 0);
                console.log(y);
    
                console.log("-------------------FIND-------------------------------");
                let z = Arr.find(x => x % 3 === 0);
                console.log(z);
    
                console.log("-------------------FILTER-------------------------------");
                let k = Arr.filter(x => x > 3);
                console.log(k);
    
    
                console.log("-------------------MAP-------------------------------");
                let a = Arr.map(x => x % 2 == 0);
                console.log(a);
    
                console.log("-------------------FLAT MAP-------------------------------");
                let b = Arr.flatMap(x => x % 2 == 0);
                console.log(b);
    
                console.log("-------------------FILL------------------------------");
                let q = Arr.fill(2, 3, 5);
                console.log(q);
    
                console.log("-------------------FLAT-------------------------------");
                var newArr = [[2, 4, [5, 6]], 7, 9, 10, [3, 6], { name: 'suji' }];
                console.log(newArr);
                let c = newArr.flat(1);
                console.log(c);
                let d = newArr.flat(2);
                console.log(d);
