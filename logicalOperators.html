AND - returns false even if 1 of the value is false; returns the LAST true value otherwise

            console.log(1 && 'Strawbery');
            console.log(1 && 2 && 5);
            console.log(1 && 'Strawbery' && false);
            console.log(false && 1);
            console.log(true && true);
            console.log(false && true);
            console.log(false && false);
            console.log(false && true);
            console.log("STRAWBERRY" && 'Strawbery');
            console.log("STRAWBERRY" && "Strawbery");
            console.log('Strawbery' && "STRAWBERRY");

            
OR - returns true even if 1 of the value is true; returns the FIRST true value otherwise

            console.log(1 || 'Strawbery');
            console.log(1 || 2 || 5);
            console.log(1 || 'Strawbery' || false);
            console.log(false || 1);
            console.log(true || true);
            console.log(false || true);
            console.log(false || false);
            console.log(false || true);
            console.log("STRAWBERRY" || 'Strawbery');
            console.log("STRAWBERRY" || "Strawbery");
            console.log('Strawbery' || "STRAWBERRY");

            
NOT - returns opposite value

            console.log(!true);
            console.log(!false);
            console.log(!false && !false);
            console.log(!true && !true);
            console.log(!true && !false);
            console.log(!false && !true);
            console.log(!false || !false);
            console.log(!true || !true);
            console.log(!true || !false);
            console.log(!false || !true);
            console.log(!1);
            console.log(!0)
            console.log(!1 && !'string') ;
