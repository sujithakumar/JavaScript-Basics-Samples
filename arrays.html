    console.log("changes to Arr1 will reflect in Arr");

    Arr1.push("push1");

    console.log(Arr);

    console.log("remove value in the last position");
    Arr1.pop();
    console.log(Arr1);

    console.log("Push value in the first position");
    Arr1.unshift("push2");
    console.log(Arr1);
    console.log(Arr);

    console.log("remove value in the first position");
    Arr1.shift("push2");
    console.log(Arr1);
    console.log(Arr);

    console.log("removing array elemnts");
    Arr1.length = 0;
    console.log(Arr1);
